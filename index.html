Underscore.string

This is a very good solution for string manipulation operations. This was originally started as an Underscore.js extension but is now a full standalone library currently at Version 3.0.3.

This Module is owned and maintained by Esa-Matti Suuronen

For those of us familiar with PHP know there are tons of string manipulation operations, but I have focused on some that may be used more frequently when developing a Node js application

Install

Install from npm



npm install underscore.string

sudo npm install underscore.string // Mac Users

Require individual functions



//using sluggify *Step 1*

var slugify = require("underscore.string/slugify"); 

//using sluggify *Step 2*

console.log(  slugify("Hello world!")  );

// => Hello-world!

or load the full library to enable chaining

chop(string, step) => array


//load the full library

var s = require("underscore.string");

// Chaining

console.log(  s("toluwalope").capitalize().chop(2)  );

//=> [To,lu,wa,lo,pe]

capitalize(string) => string


console.log(  s("tolu rocks").capitalize().value()  );

//=> Tolu rocks

isBlank(string) => boolean


console.log(  s("").isBlank()   );

//=> True

join(separator, ...strings) => string


console.log(  s("Pneumonoultramicroscopicsilicovolcanoconiosis").chop(8).join("_")  );

//=> Pneumono_ultramic_roscopic_silicovo_lcanocon_iosis

reverse(string) => string


console.log(  s("Love").reverse().value()  );

//=> evoL

humanize(string) => string


console.log(  s("  I dont-pop_molly I_Rock-tomFord  ").humanize().value() );

//=> I dont pop molly i rock tom ford
